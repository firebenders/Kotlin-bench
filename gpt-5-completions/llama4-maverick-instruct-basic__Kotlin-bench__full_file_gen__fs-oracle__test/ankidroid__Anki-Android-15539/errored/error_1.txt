### Step 6: Handle user actions in the Reviewer

We update the `Reviewer` class to handle the user actions.

[start of AnkiDroid/src/main/java/com/ichi2/anki/Reviewer.kt]
```kotlin
override fun executeCommand(which: ViewerCommand, fromGesture: Gesture?): Boolean {
    when (which) {
        // ...
        ViewerCommand.USER_ACTION_1 -> {
            executeUserAction(1)
            return true
        }
        ViewerCommand.USER_ACTION_2 -> {
            executeUserAction(2)
            return true
        }
        ViewerCommand.USER_ACTION_3 -> {
            executeUserAction(3)
            return true
        }
        ViewerCommand.USER_ACTION_4 -> {
            executeUserAction(4)
            return true
        }
        ViewerCommand.USER_ACTION_5 -> {
            executeUserAction(5)
            return true
        }
        ViewerCommand.USER_ACTION_6 -> {
            executeUserAction(6)
            return true
        }
        ViewerCommand.USER_ACTION_7 -> {
            executeUserAction(7)
            return true
        }
        ViewerCommand.USER_ACTION_8 -> {
            executeUserAction(8)
            return true
        }
        else -> return super.executeCommand(which, fromGesture)
    }
}

private fun executeUserAction(action: Int) {
    val js = sharedPrefs().getString("userAction$action", null) ?: return
    Timber.d("executeUserAction($action): $js")
    loadUrlInViewer("javascript:$js")
}
```
### Step 7: Update the AbstractFlashcardViewer

We update the `AbstractFlashcardViewer` class to handle the user actions.

[start of AnkiDroid/src/main/java/com/ichi2/anki/AbstractFlashcardViewer.kt]
```kotlin
fun loadUrlInViewer(url: String) {
    processCardAction { cardWebView: WebView? -> cardWebView!!.loadUrl(url) }
}
```
