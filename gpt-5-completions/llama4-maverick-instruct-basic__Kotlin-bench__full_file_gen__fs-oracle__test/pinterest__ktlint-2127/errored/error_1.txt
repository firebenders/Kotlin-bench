To solve the given issue, we need to modify the `IndentationRule` and `WrappingRule` classes in the `ktlint-ruleset-standard` module.

The issue is related to the indentation of expressions wrapped in parentheses. The current implementation of the `IndentationRule` does not correctly handle this case.

Here are the steps to fix the issue:

1.  Modify the `IndentationRule` to correctly indent expressions wrapped in parentheses.
2.  Update the `WrappingRule` to rearrange the parentheses when necessary.

Here are the modified files:

package com.pinterest.ktlint.ruleset.standard.rules

// ...

public class WrappingRule :
    StandardRule(
        id = "wrapping",
        usesEditorConfigProperties =
            setOf(
                INDENT_SIZE_PROPERTY,
                INDENT_STYLE_PROPERTY,
                MAX_LINE_LENGTH_PROPERTY,
            ),
        visitorModifiers =
            setOf(
                RunAfterRule(
                    ruleId = ANNOTATION_RULE_ID,
                    mode = REGARDLESS_WHETHER_RUN_AFTER_RULE_IS_LOADED_OR_DISABLED,
                ),
            ),
    ) {
    // ...

    override fun beforeVisitChildNodes(
        node: ASTNode,
        autoCorrect: Boolean,
        emit: (offset: Int, errorMessage: String, canBeAutoCorrected: Boolean) -> Unit,
    ) {
        when (node.elementType) {
            // ...

            LPAR -> rearrangeBlock(node, autoCorrect, emit)

            // ...
        }
    }

    private fun rearrangeBlock(
        node: ASTNode,
        autoCorrect: Boolean,
        emit: (offset: Int, errorMessage: String, canBeAutoCorrected: Boolean) -> Unit,
    ) {
        // ...

        if (node.treeParent.elementType == PARENTHESIZED) {
            // rearrange:
            //   val foo = (
            //       bar1 +
            //           bar2
            //   )
            // to:
            //   val foo = (
            //       bar1 +
            //       bar2
            //   )
            val r = node.nextSibling { it.elementType == RPAR }!!
            if (!node.nextCodeLeaf().isWhiteSpaceWithNewline()) {
                requireNewlineAfterLeaf(node, autoCorrect, emit)
            }
            if (!r.prevLeaf().isWhiteSpaceWithNewline()) {
                requireNewlineBeforeLeaf(r, autoCorrect, emit, indentConfig.parentIndentOf(node))
            }
        }
    }

    // ...
}
